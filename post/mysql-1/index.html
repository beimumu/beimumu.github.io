-- 1、查询所有学生的学号、姓名，要求按学号升序排列
/*
升序排列：
where 所要求的 order by 字段 [asc升序 | desc降序] 
*/
-- 方法一
select Sno,Sname from Student
where Sno order by Sno;
-- 方法二
select Sno,Sname from Student order by Sno asc;

-- 2、将“060101”号学生的成绩按降序排列
select Cno,Grade from Sc
where Sno='060101' ORDER BY Grade Desc;

-- 3、查询课程名中不包含“数据”字样的课程信息
-- 不包含 not like '字段'
select * from course
where Cname not like '%数据%'

-- 4、查询选修了100004号课程且成绩在80分以上的学生学号、姓名和成绩
/*
Sname只有在Student表中有，Grade只有在SC表中有，其中将两个表连起来的是Sno。
要得到Student表中的sno,就得通过sc表中的sno,所以Student.Sno=sc.Sno,
即：
select Student.Sname,Student.Sno,SC.Cno,SC.Grade from SC,Student
where Student.Sno = SC.Sno and sc.Cno = '100004' and sc.Grade >= '80';
*/
select Student.Sname,Student.Sno,SC.Cno,SC.Grade from SC,Student
where Student.Sno = SC.Sno and sc.Cno = '4' and sc.Grade >= '80';

-- 5、查询选修课程“100002”的学生中成绩最高的学生的学号
/*
先查询Sno，在嵌套查询成绩最高
列值的最大值: max()
列值的最小值：min()
即：
select Sno from sc
where Grade=(select MAX(Grade) from sc where Cno = '100002' );
*/
select Sno 
from sc
where Grade=(select MAX(Grade) from sc where Cno = '4' );

-- 查询选修课程“4”的学生中成绩最高的学生的学号
select sno 
from SC
where cno = 4
order by grade desc
limit 0,1;

-- 查询课程号“4”的最高分和最低分
select max(Grade) as '最高分', min(Grade) as '最低分' from sc
where Cno = '4';
-- 查询选修课程“100002”的学生中成绩最高的学生的学号、姓名和成绩
select student.Sname,sc.Sno,sc.Grade from student,sc
where student.Sno = sc.Sno and Grade = (select MAX(Grade) from sc where Cno = '4' );

-- 6、查询所有比王华年龄大的学生姓名、年龄和性别
/*
子查询：
先在student表中找王华的年龄，
再在student表中找大于王华年龄的姓名...
即：
select Sname,Sage,Ssex from student
where  sage > (select sage from student where sname = '王华')
*/
select Sname,Sage,Ssex from student
where  sage > (select sage from student where sname = '钟文辉')  

-- 查询选了“4”号课程的且成绩高于此课程平均成绩的学生的学号、该门课程成绩
/*
先查出此课程（课程号为4）的平均成绩
select avg(grade) 
from sc
where cno = '4'
再查出4号的学生学号和成绩，
select sno,grade
from sc
where cno = '4'
*/
select sno,grade
from sc
where cno = '4' and grade > (select avg(grade)
from sc where cno = '4');

-- 7、查询学号为“16480101”学生的成绩，输出格式为：姓名、课程名、成绩
/*
在select中只需要输入你想查询的内容，不用输那么多；
在条件中，你可以调用表中的任何字段。其中，
三表相联
student.Sno = sc.Sno and course.Cno = sc.Cno 相当于关联。
*/

-- 学生表和成绩表的姓名对应，课程表和成绩表的课程名没对应，会导致该学号下的同学课程名和成绩不匹配。
select student.Sname as '姓名',course.Cname as '课程名',sc.Grade as '成绩' from student,course,sc
where student.Sno = sc.Sno and sc.Sno = '060101' 
-- 修改后：
-- 方式一: 在查询语句里，表明调用导致代码冗余
select student.Sname as '姓名',course.Cname as '课程名',sc.Grade as '成绩' 
from student ,course ,sc
where student.Sno = sc.Sno and course.Cno = sc.Cno and sc.Sno = '060101' ;

-- 方式二：方式一精简，取掉表用调用
select Sname as '姓名',Cname as '课程名',Grade as '成绩' 
from student s,course c,sc
where s.Sno = sc.Sno and c.Cno = sc.Cno and sc.Sno = '060101' ;

-- 方式三：join
select Sname,cname,Grade
from student s
inner join sc
on s.sno = sc.Sno
inner join course c
on sc.Cno = c.Cno
where sc.Sno = 060101;

-- 8、查询本班同学各科的平均成绩，输出格式为：学号、姓名、平均成绩，并按降序排列
/*
“每个”等字样，考虑group by子句
语法：group by [分组依据]
*/
select s.sno as '学号',sname as '姓名',avg(grade) as '平均成绩'
from student s
inner join sc
on s.Sno = sc.Sno
group by sc.sno
order by avg(grade) desc;
